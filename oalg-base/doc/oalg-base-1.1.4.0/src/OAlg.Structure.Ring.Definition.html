-- Semiring -

--------------------------------------------------------------------------------
-- Ring -

-- | abelian semi rings.
type Ring r = (Semiring r, Abelian r)

--------------------------------------------------------------------------------
-- Galoisian -

-- | __Note__ Not every element not equal to 'zero' has to be invertible. As
--   such 'Z' is 'Galoisian'.
type Galoisian r = (Ring r, Commutative r, Invertible r)
