-- Embeddable -

-- | canonical embedding from @__a__@ in to @__b__@.

-- __Property__

--

--     @'prj' '.' 'inj'@ is the identical mapping.

  -- | canonical injetion from @a@ in to @b@.
  inj :: a -> b
